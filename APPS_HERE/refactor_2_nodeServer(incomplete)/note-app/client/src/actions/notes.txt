export function fetchPosts() {
 const request = axios.get(`${ROOT_URL}/posts`);
return {
 type: FETCH_POSTS,
 payload: request
 };
}


export function fetchPostsSuccess(posts) {
 return {
 type: FETCH_POSTS_SUCCESS,
 payload: posts
 };
}


export function fetchPostsFailure(error) {
 return {
 type: FETCH_POSTS_FAILURE,
 payload: error
 };
}
